define(function () {

var r = '[\\$\\(\\)\\*\\+\\.\\[\\]\\?\\\\\/\\^\\{\\}\\|]',
	R = RegExp,
	e = new R(r, 'g'),
	d = new R('\\\\(?=' + r + ')', 'g');

return {
	encode: function(s) {
		return ('' + s).replace(e, '\\$&');
	},
	decode: function(s) {
		return ('' + s).replace(d, '');
	}
};
	
});
