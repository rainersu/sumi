define([
	'./isInteger',
	'./MAX_SAFE_INTEGER'
], function (I, S) {
	
return Number.isSafeInteger = Number.isSafeInteger || function(v) {
	return I(v) && v >= -S && v <= S;
};

});

* ECMAScript 6 Draft <br>
* 判断指定变量是否是安全的整数。

* @access public

* @param {*} val - 任意值
* @returns {boolean}
* @example
* console.log(Number.isSafeInteger(3));                    // true
