define(function () {
	
return {
	camelCase: function (s) {
		return ('' + s).replace(/\W+([a-z])/ig, function (a, b) {
			return b.toUpperCase();
		});
	},
	kebabCase: function (s) {
		return ('' + s).replace(/(?:\b|[a-z0-9])(?=[A-Z])/g, '$&-').toLowerCase();	
	}
};

});
