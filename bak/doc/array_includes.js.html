define(function () {

return Array.prototype.includes = Array.prototype.includes || function (searchElement, fromIndex) {
    var o = Object(this),
		l = o.length >>> 0,
		n = fromIndex >> 0,
		k = n < 0 ? Math.max(l + n, 0) : n,
		s = searchElement,
		i;
    while(k < l) { 
		if(k in o) {
			i = o[k];
			if (s === i || (s !== s && i !== i)) {
				return true;
			}
		}
		k++;
	}
    return false;
};

});
