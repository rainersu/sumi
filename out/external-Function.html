Number.prototype.fix = Number.prototype.toFixed.memoize();
console.log((2).fix(3));             // 2.000

Number.prototype.fix = (function(len) {
    return this.toFixed(len);
}).memoize();
console.log((2).fix(5));             // 2.00000

var max = Math.max.memoize(3);
console.log(max(6));                 // 6
console.log(max(2));                 // 3
