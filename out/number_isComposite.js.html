define([
	'./isInteger'
], function (I) {
	
return Number.isComposite = function(val) {
	var n = 2,
		v = +val,
		q;
	if(I(v) && v > 3) {
		q = Math.sqrt(v);
		while(n <= q) {
			if(v % n++ < 1) return true;
		}
	}
	return false;
};

});
