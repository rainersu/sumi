define(function () {

return Array.prototype.filter = Array.prototype.filter || function(func, that) {
	for(var i = 0, o = Object(this), l = o.length >>> 0, a = [], v; i < l; i++) {
		if (i in o && func.call(that, v = o[i], i, o)) a.push(v);
	}
	return a;		
};

});
