define([
	'../var/undefined'
], function (U) {

return Array.prototype.fill = Array.prototype.fill || function(value, start, end) {
	var o = Object(this),
		l = o.length >>> 0,
		r = start >> 0,
		m = Math,
		k = r < 0 ? M.max(l + r, 0) : M.min(r, l),
		d = end === U ? l : end >> 0,
		f = d < 0 ? M.max(l + d, 0) : M.min(d, l);
	while(k < f) o[k++] = value;
	return o;	
};

});
