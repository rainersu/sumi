define([
	'./number',
	'../math/clip',
	'../misc/classof'
], function (R, P, C) {

return function (o, m) {
	var b = C(o) === 'string',
		t = Object(b ? o.split('') : o),
		l = t.length >>> 0,
		r = new Array(l),
		i = 0,
		n;
	for(; i < l; i++) {
		n = R(0, i);
		if (n != i) r[i] = r[n];
		r[n] = t[i];
	}
	r = m ? r.slice(P(l - m, l - 1)) : r;
	return b ? r.join('') : r;
};

});
