define([
	'../number/MAX_SAFE_INTEGER',
	'../number/MIN_SAFE_INTEGER',
	'../var/undefined'
], function (X, N, U) {
	
return function(min, max, int) {
	var x = max == U ? X : +max,
		n = min == U ? N : +min,
		i = !(int - 1),
		r = n + Math.random() * (x - n);
	return  i ? Math.floor(r) : r;
};

});

/**
* 生成一个大于或等于 min 并小于 max 的随机数。
* @access public
* @func rnd.num
* @param {number} [min=Number.MIN_SAFE_INTEGER] - 范围下限（被包括）
* @param {number} [max=Number.MAX_SAFE_INTEGER] - 范围上限（不包括）
* @param {boolean} [int = true] - 是否需要整数
* @returns {number} 
* @example
* console.log(rnd.num());
* console.log(rnd.num(1));
* console.log(rnd.num(1, 10));
* console.log(rnd.num(1, 10, false));
