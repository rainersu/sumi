var f = 'function',
	F = 'constructor',
	c = 'boolean number string';

function m (a, b) {
	for  (var i in a) {
		if (H(a, i) !== H(b, i) || !e(a[i], b[i])) return 1;
	}
	return 0;
}

function e(a, b) {
	if (a !== b) {
		var x = C(a),
			y = C(b);
		if (
			x === f ||
			y === f ||
			x !== y ||
			a[F] !== b[F] ||

			m(a, b) ||

		}
	}

}
