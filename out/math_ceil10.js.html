define(function () {
	
return Math.ceil10 = Math.ceil10 || function(value, exp) {
	var v = +value,
		e = exp >> 0,
		f = Math.ceil,
		g = 'e';
	if(!e) {
		return f(v);
	}
	v = v.toString().split(g);
	v = f(+(v[0] + g + (v[1] ? (+v[1] - e) : -e))).toString().split(g);
	return +(v[0] + g + (v[1] ? (+v[1] + e) : e));
};

});
