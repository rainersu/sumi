define([
	'../misc/classof',
	'./number'
], function (C, N) {
	
return function(obj, min, max) {
	var o = Object(obj),
		l = o.length >>> 0;
	if (l) {
		var n = N((max || o.length) >>> 0, min >>> 0);
		return C(o) === 'string' ? o.charAt(n) : o[n];
	}
};

});
