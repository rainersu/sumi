define(
[
	'./isObject',
	'../var/hasOP'
]
, function (
	isObject,
	hasOP
) 
{'use strict';

return function (v) {
	var s = 'constructor',
		c;
	if (!v || !isObject(v) || v.nodeType || v === v.window) {
		return false;
	}
	try {
		c = v[s];
		if (c && !hasOP.call(v, s) && !hasOP.call(c.prototype, "isPrototypeOf")) {
			return false;
		}
	} 
	catch (e) {
		return false;
	}
	for (s in v) {
		if (!hasOP.call(v, s)) return false;
	}
	return true;
};

});
