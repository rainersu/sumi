define(
[
	'../var/slice',
	'./stringify',
	'../var/propIE'
]
, function (
	slice,
	stringify,
	propIE
) 
{'use strict';
	
return function (t, m) {
	var c = {};
	return function () {
		var a =  slice.call(arguments),
			k =  stringify(a, m);
		return propIE.call(c, k) ? c[k] : c[k] = t.apply(t, a);
	}
};

});

* @param {number}  mode.0  - 使用 Javascript 的自动类型转换，相当于 String(val)

* @returns {function} 创建出的新函数
