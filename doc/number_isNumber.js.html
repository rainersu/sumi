* @returns {boolean}
* @example
* console.log(Sumi.Number.isNumber(NaN));                   // true
* console.log(Sumi.Number.isNumber(Infinity));              // true

* console.log(Sumi.Number.isNumber(Number('3.')));          // true
* console.log(Sumi.Number.isNumber(new Number('.3')));      // true
