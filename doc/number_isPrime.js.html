define([
	'./isInteger'
], function (I) {
	
return Number.isPrime = function(val) {
	var n = 2,
		v = +val,
		q;
	if(!I(v) || v < 2) return false;
	q = Math.sqrt(v);
	while (n <= q) {
		if(v % n++ < 1) return false;
	}
	return true;
};

});
