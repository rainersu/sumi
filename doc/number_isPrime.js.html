define(
[
	'./ECMA/isInteger'
]
, function (
		isInteger
) 
{'use strict';
	
return function (v) {
	var n = 2,
		q;
	v = +v;
	if(!isInteger(v) || v < 2) return false;
	q = Math.sqrt(v);
	while (n <= q) {
		if(v % n++ < 1) return false;
	}
	return true;
};

});
