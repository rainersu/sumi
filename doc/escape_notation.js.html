define(function () {'use strict';
	
return {
	camelCase: function(s) {
		return ('' + s).replace(
			/^-ms-/i, 'ms-'
		).toLowerCase().replace(
			/(^[^a-z0-9-]+|[^a-z0-9]+$)|[^a-z0-9]+([a-z0-9])/ig, 
			function (a, b, c) {
				return b ? '' : c.toUpperCase();
			}
		);
	},
	kebabCase: function(s) {
		return ('' + s).replace(
			/(^[^A-Za-z0-9-]+|[^A-Za-z0-9]+$)|(?:^|[a-z0-9])(?=[A-Z])|([^A-Za-z0-9]+)/g, 
			function (a, b, c) {
				return b ? '' : c ? '-' : a + '-';
			}
		).toLowerCase().replace(
			/^ms-/, '-ms-'
		);
	},
	snakeCase: function(s, b) {
		return ('' + s).replace(
			/(^[^A-Za-z0-9]+|[^A-Za-z0-9]+$)|[a-z0-9](?=[A-Z])|([^A-Za-z0-9]+)/g, 
			function (a, b, c) {
				return b ? '' : c ? '_' : a + '_';
			}
		).toUpperCase();
	}
};

});
