		m = Math,
		l = o.length >>> 0,
		k = r < 0 ? m.max(l + r, 0) : m.min(r, l),
		d = arguments.length < 3 ? l : g >> 0,
		f = d < 0 ? m.max(l + d, 0) : m.min(d, l);
	while(k < f) o[k++] = v;
	return o;
