define(function () {

function f(n) {
	return n < 10 ? '0' + n : n;
};

return Date.prototype.toISOString = Date.prototype.toISOString || function() {
	var t = this;
	return t.getUTCFullYear() +
		'-' + f(t.getUTCMonth() + 1) +
		'-' + f(t.getUTCDate()) +
		'T' + f(t.getUTCHours()) +
		':' + f(t.getUTCMinutes()) +
		':' + f(t.getUTCSeconds()) +
		'.' + (t.getUTCMilliseconds() / 1000).toFixed(3).slice(2, 5) +
		'Z';
};

});
