define(
[
	'../math/rnd',
	'../var/slice'
]
, function (
	rnd,
	slice
) 
{'use strict';

return function (t, m) {
	t = slice.call(Object(t));
    var l = t.length >>> 0,
		p, 
		w;
    while(l--) { 
		p = rnd(0, l); 
		w = t[l]; 
		t[l] = t[p]; 
		t[p] = w;
	}   
    return m ? t.slice(0, m) : t;
};

});
