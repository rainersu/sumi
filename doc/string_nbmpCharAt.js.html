define(function () {'use strict';
	
return function(s, i) {
	s = '' + s;
	i = i >> 0;
	var r = '',
		l = s.length,
		u = /[\uD800-\uDBFF][\uDC00-\uDFFF]/g,
		x = /[\uD800-\uDBFF]/,
		y = /[\uDC00-\uDFFF]/,
		c;
	while(u.exec(s)) {
		if (u.lastIndex - 2 < i) i++;
		else break;
	}
	if(i > 0 && i < l) {
		r += s.charAt(i);
		c = s.charAt(i + 1);
		if(x.test(r) && y.test(c)) {
			r += c;
		}
	}
	return r;
};

});
