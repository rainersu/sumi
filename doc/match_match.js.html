define(
[
	'../core/cp',
	'../core/type',
	'../var/hasOP'
]
, function (
	cp,
	type,
	hasOP
) 
{'use strict';

var C = {};

function Match (p, k, f) {
	this.source  =   p;
	k = !k  ? [] : type(k) === 'string' ? k.split(/[\s,;|]+/) : Array(k);
	k.unshift('match');
	this.keys    =   k;
	this.parser  =   f;
}

cp(
	Match.prototype, 
	{
		build  : function (y, g) {
			return R(this.source, y, g);
		},
		is     : function (s) {
			return this.build(2).test( s );
		},
		in     : function (s, b) {
			return this.build(b).test( s );
		},
		index  : function (s, b) {
			return s.search(this.build( ));
		},
		match  : function (s, b) {
			return s.match(this.build(b, 'g')) || [];
		},
		parse  : function (s, b, f) {
			f = f || this.parser; 
			var r = [],
				k = this.keys,		
				e = this.build(b, 'g'),
				l,
				a,
				o,
				v,
				j;
			while ((a = e.exec(s)) !== null) {
				o = {};
				for ( l = k.length; l--; ) {
					v = a[l];
					if (o[j = k[l] || l] === undefined && v !== undefined) o[j] = v;
				}
				if (f) {
					v = f(o);
					if (v !== undefined) o = v;
				}
				r.push(o);
			}
			return r;
		},
		replace : function (s, f, b, g) {
			return s.replace(this.build(b, g || 'g'), f);
		}
	}
);

cp(
	Match, 
	{
		build   : R,
		clear   : function () { C = {}; },
		cache   : true
	}
);

function R (p, y, g) {
	y = y |  0;
	g = g ||'';
	p = type(p) === 'regexp' ? p.source :  p;
	p = (y > 1 ? '^(?:' + p + ')$' : y > 0 ? '(?:^|\\b)(?:' + p + ')(?:$|\\b)' : p);
	y = '/'  + p +  '/' + g;
	return C[y] || (C[y] = new RegExp(p, g));
}

return Match;

});

/**

* @see {@link Sumi.Match#match}
