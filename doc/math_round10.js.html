define(function () {'use strict';
	
return function (v, e) {
	v = +v;
	e = e >> 0;
	if(!e) return Math.round(v);
	var g = 'e';
	v = v.toString().split(g);
	v = Math.round(+(v[0] + g + (v[1] ? (+v[1] - e) : -e))).toString().split(g);
	return +(v[0] + g + (v[1] ? (+v[1] + e) : e));
};
