define(function () {

return Array.prototype.find = Array.prototype.find || function(func, that) {
	for(var i = 0, o = Object(this), l = o.length >>> 0, v; i < l; i++) {
		if (i in o && func.call(that, v = o[i], i, o)) return v;
	}	
};

});
