define([
	'../var/undefined',
	'../var/object'
], function (U, O) {

return Array.prototype.reduceRight = Array.prototype.reduceRight || function(cb, iv) {
	var o = O(this), 
		l = (o.length >>> 0) - 1, 
		b = 1 < arguments.length, 
		v = b ? iv : U;
	for(; l + 1; l--) {
		if(l in o) {
			if(b) {
				v = cb(v, o[l], l, o);
			}
			else {
				v = o[l];
				b = true;
			}
		}
	}
	return v;	
};

});
