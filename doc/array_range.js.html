return function (u, v, s) {
	var b = arguments.length > 1,
		z = u >>  0,
		x = b ? z : 0,
		y = b ? v >> 0 : z,
		m = Math.abs,
		a = m(s >> 0 || 1),
		r = [],
		n = (y - 1 - x) / a >> 0,
		i = 0;
	if (n < 0) a = -a;
	n = m(n);
	while (i <= n) {
		r[i++] = x;
		x+= a;
	}
	return r;
};
