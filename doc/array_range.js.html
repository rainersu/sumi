define(function () {

return Array.range = function (min, max, step) {
	var b = arguments.length > 1,
		z = min >>  0,
		x = b ? z : 0,
		y = b ? max >> 0 : z,
		a = Math.abs(step >> 0 || 1),
		r = [],
		n = (y - 1 - x) / a >> 0,
		i = 0;
	if (n < 0) a = -a;
	n = Math.abs(n);
	while (i <= n) {
		r[i++] = x;
		x+= a;
	}
	return r;
};

});

* @param {number} [min=0] - 数字序列的起始值（被包括）。如果没有指定 max ，则作为 max 值，并将 0 作为 min 值
* @param {number} [max] - 数字序列的上限值（不包括）
* @param {number} [step=1] - 步进增量
