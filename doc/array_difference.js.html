define(
[
	'../var/slice',
	'../var/concat',
	'./uniq'
]
, function (
	slice, 
	concat,
	uniq
) 
{'use strict';
	
return function () {
	var a = slice.call(arguments),
		o = uniq(a.shift()),
		l = o.length,
		c = concat.apply([], a),
		h = c.length,
		r = [],
		i = 0,
		m,
		n,
		w,
		x,
		y,
		b;
	if (!h) r = o;
	else if (l) {
		for(; i < l; i++) {
			b = true;
			m = o[i];
			for(n = 0; n < h; n++) {
				w = c[n];
				if (w === m || (w !== w && m !== m)) {
					b = false;
					break;
				}	
			}
			if (b) r.push(m);
		}
	} 
	return r;
};

});
