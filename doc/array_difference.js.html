define([
	'../var/slice',
	'../var/concat',
	'./unique'
], function (S, A, Q) {
	
return Array.difference = function () {
	var a = S(arguments),
		o = Q.call(a.shift()),
		l = o.length,
		c = A([], a),
		h = c.length,
		r = [],
		i = 0,
		m,
		n,
		w,
		x,
		y,
		b;
	if (!h) r = o;
	else if (l) {
		for(; i < l; i++) {
			b = true;
			m = o[i];
			for(n = 0; n < h; n++) {
				w = c[n];
				if (w === m || (w !== w && m !== m)) {
					b = false;
					break;
				}	
			}
			if (b) r.push(m);
		}
	} 
	return r;
};

});
