define(function () {'use strict';

return function (r, s, g) {
	r = r >> 0;
	s = s >> 0;
	var m = Math,
		a = Object(this),
		l = a.length >>> 0,
		t = r < 0 ? m.max(l + r, 0) : m.min(r, l),
		f = s < 0 ? m.max(l + s, 0) : m.min(s, l),
		e = arguments.length < 3 ? l : g >> 0,
		n = e < 0 ? m.max(l + e, 0) : m.min(e, l),
		c = m.min(n - f, l - t),
		d = 1;
	if (f < t && t < (f + c)) {
		d = -1;
		f+= c - 1;
		t+= c - 1;
	}
	while (c > 0) {
		if (f in a) a[t] = a[f];
		else delete a[t];
		f+= d;
		t+= d;
		c--;
	}
	return a;
};

});
