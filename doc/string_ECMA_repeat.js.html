define(
[
	'../../number/ECMA/isSafeInteger'
]
, function (
	isSafeInteger
) 
{'use strict';
	
return function (count) {
	var c = count >> 0,
		r = '',
		s = r + this;
	if(s.length && isSafeInteger(c) && c > 0) for(;;) {
		if((c & 1) === 1) {
			r += s;
		}
		c >>>= 1;
		if(!c) break;
		s+= s;
	}
	return r;
};

});
