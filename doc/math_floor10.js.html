return function (v, e) {
	v = +v;
	e = e >> 0;
	if(!e) return Math.floor(v);
	var g = 'e';
	v = v.toString().split(g);
	v = Math.floor(+(v[0] + g + (v[1] ? (+v[1] - e) : -e))).toString().split(g);
	return +(v[0] + g + (v[1] ? (+v[1] + e) : e));
};
