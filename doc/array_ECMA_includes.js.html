define(
[
	'../../core/eq'
]
, function (
	eq
) 
{'use strict';

return Array.prototype.includes = Array.prototype.includes || function (s, f) {
    var o = Object(this),
		l = o.length,
		n = f >> 0,
		k = n <  0 ? Math.max(l + n, 0) : n;
    while  (k <  l) { 
		if (k in o && eq(s, o[k])) return true;
		k++;
	}
    return false;
};

});
