define(
[
	'../core/eq',
	'./reorder'
]
,function (
	eq,
	reorder
) 
{'use strict';

return function (a) {
	a = reorder(a);
	var r = [],
		l = a.length,
		i = 1;
	if (l) {
		r.push(a[0]);
		for(; i < l; i++) {
			if (i in a && !eq(a[i - 1], a[i])) {
				r.push(a[i]);
			}
		}
	}
	return r;
};

});

* console.log(Sumi.Array.uniq(['1', '1', '2', 1, '3', '3', 2, '2', '1', 3, NaN, NaN]));
* // [ 1, '3', '1', '2', 3, NaN, 2 ]
* 
* console.log(Sumi.Array.uniq([2, '1', 1, true, 1.0]));

*/
