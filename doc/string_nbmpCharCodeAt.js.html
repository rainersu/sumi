define(function () {
	
return String.prototype.nbmpCharCodeAt = function(idx) {
	var s = this,
		i = idx >> 0,
		c = s.charCodeAt(i),
		h, 
		l;
	if(0xD800 <= c && c <= 0xDBFF) {
		h = c;
		l = s.charCodeAt(i + 1);
		return isNaN(l) ? NaN : ((h - 0xD800) * 0x400) + (l - 0xDC00) + 0x10000;
	}
	if(0xDC00 <= c && c <= 0xDFFF) {
		return NaN;
	}
	return c;
};

});
