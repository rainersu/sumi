{'use strict';

return  function (s, f) {
	var o = Object(this),
		l = o.length >>> 0,
		n = f >> 0,
		k = Math.max(n < 0 ? l + n : n, 0),
		i;
    while (k <  l) { 
		if(k in o && eq(s, o[k])) return k;
		k++;
	}
    return -1;
};

});

* @example
* console.log([1, 2, 3].indexOf(2));                       //  1
