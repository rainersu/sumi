define(
[
	'./type'
]
, function (
	type
) 
{'use strict';

return function (a, b) {
	var c, d;
	return a === b || (a !== a && b !== b) ? 0 :
	       a !== a   && b === b   ? -1 :
	       a === a   && b !== b   ?  1 :
	       a != null && b == null ?  1 :
	       a == null && b != null ? -1 : 
	      (c = type(a), d = type(b), c === d) ? (a > b ? 1 : -1) : 
	      (c > d ? 1 : -1);
};

});
