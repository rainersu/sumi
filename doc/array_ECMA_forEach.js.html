		l = o.length,
		i = 0;
	for(t = t || o; i < l; i++) {
		if (i in o) f.call(t, o[i], i, o);

* @returns {undefined}
* @example
* var a = [1, 2, 3];
* var f = function (n, i) { a[i] = ++n; };
* a.forEach(f);

* var f = function (n) { a += n; };
