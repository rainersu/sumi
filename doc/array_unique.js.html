define([
	'../json/json'
], function (J) {

return Array.prototype.unique = function() {
	var a = Object(this),
		o = {},
		r = [],
		l = a.length >>> 0,
		i = 0,
		s = J.stringify,
		m,
		n;
	for(; i < l; i++) {
		m = a[i];
		n = s(m);
		if(!o[n]) {
			o[n] = 1;
			r.push(m);
		}
	}
	return r;
};

});
