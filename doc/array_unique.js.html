define(
[
	'../var/slice',
	'../core/json'
]
,function (
	slice,
	JSON
) 
{'use strict';

return function (a) {
	a = slice.call(Object(a));
	var l = a.length;
	if (l < 2) return a;
	var s = JSON.stringify,
		o = {},
		r = [],
		i = 0,
		m,
		n;
	for(; i < l; i++) if (i in a) {
		m = a[i];
		n = s(m);
		if(!o[n]) {
			o[n] = 1;
			r.push(m);
		}
	}
	return r;
};

});
