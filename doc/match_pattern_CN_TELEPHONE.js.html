define({
	name   : 'CN_TELEPHONE',
	pattern: '(?:\\+86[- ](\\d{2,3})|\\(\\+?86\\)[- ]?(\\d{2,3})|\\(\\+?86[- ](\\d{2,3})\\)|\\(0(\\d{2,3})\\)|0(\\d{2,3}))[- ]?(\\d{3,4}[- ]?\\d{4})',
	keys   : 'zone,zone,zone,zone,zone,number',
	parser : function (o) {
		o.number = o.number.replace(/[-\s]+/g, '');
		o.zone = +o.zone;
		return o;
	}
});
