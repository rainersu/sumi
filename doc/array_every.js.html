define(function () {

return Array.prototype.every = Array.prototype.every || function(f, t) {
	for(var o = Object(this), l = o.length >>> 0; l--;) {
		if(l in o && !f.call(t, o[l], l, o)) return !!0;
    }
	return !!1;
};

});
