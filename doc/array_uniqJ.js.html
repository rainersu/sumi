var s = JSON.stringify;

return function (a) {
	a = Object(a);
	var o = {},
		r = [],
		l = a.length >>> 0,
		i = 0,
		m,
		n;
	for(; i < l; i++) if (i in a) {
		m = a[i];
		n = s(m);
		if(!o[n]) {
			o[n] = 1;
			r.push(m);
		}
	}
	return r;
};

});
