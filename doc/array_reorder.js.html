function f(a, b) {
	var c = typeof a,
		d = typeof b;
	return a === b || (a !== a && b !== b) ? 0 : c === d ? (a < b ? -1 : 1) : (c < d ? -1 : 1); 
}

var a = {
		'object'  : 1,
		'function': 1
	},
	sort = Array.prototype.sort;
	
return function (w) {
	var u = slice.call(arguments, 1),
		l = u.length;
	return sort.call(slice.call(Object(w)), l ? (function() {	
		return function(o, p) {	
			var m,
				i = 0,
				r;
			if(o && p && typeof o in a && typeof p in a) while (i < l && !r) {
				m = u[i++];
				r = f(o[m], p[m]);	
			}				
			return r;
		};
	})() : f);
};
