define(function () {

function f(a, b) {
	var c = typeof a,
		d = typeof b;
	return a === b ? 0 : c === d ? (a < b ? -1 : 1) : (c < d ? -1 : 1); 
}

var a = {
		'object'  : 1,
		'function': 1
	};
	
return Array.prototype.reorder = function() {
	var u = arguments,
		l = u.length;
	return this.sort(l ? (function() {	
		return function(o, p) {	
			var m,
				i = 0,
				r;
			if(o && p && typeof o in a && typeof p in a) while (i < l && !r) {
				m = u[i++];
				r = f(o[m], p[m]);	
			}				
			return r;
		};
	})() : f);
};

});
