return   function (s, f) {
	var o = Object(this),
		l = o.length >>> 0,
		n = arguments.length < 2 ? l :  f >> 0,
		k = Math.min(n < 0 ? l + n : n, l - 1),
		i;
    while  (k > -1) { 
		if (k in o && eq(s, o[k])) return k;
		k--;
	}
    return -1;
};
