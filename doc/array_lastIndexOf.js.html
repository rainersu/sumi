define(function () {

return Array.prototype.lastIndexOf = Array.prototype.lastIndexOf || function (searchElement, fromIndex) {
	var o = Object(this),
		l = o.length >>> 0,
		n = arguments.length < 2 ? l : fromIndex >> 0,
		k = Math.min(n < 0 ? l + n : n, l - 1),
		s = searchElement,
		i;
    while(k > -1) { 
		if(k in o) {
			i = o[k];
			if (s === i || (s !== s && i !== i)) {
				return k;
			}
		}
		k--;
	}
    return -1;
};

});
