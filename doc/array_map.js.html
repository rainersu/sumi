define(function () {

return Array.prototype.map = Array.prototype.map || function(func, that) {
	for(var i = 0, o = Object(this), l = o.length >>> 0, a = new Array(l); i < l; i++) {
		if (i in o) a[i] = func.call(that, o[i], i, o);
    }
	return a;		
};

});
