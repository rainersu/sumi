define(function () {
	
return String.prototype.codePointAt = String.prototype.codePointAt || function(idx) {
   	var s = this,
		i = idx >> 0,
		l = s.length,
		f = s.charCodeAt(i),
		w = s.charCodeAt(i + 1),
		e = (i + 1 === l);
   if(i > -1 && i < l) {
	   return f < 0xD800 || f > 0xDBFF || e || w < 0xDC00 || w > 0xDFFF ? f : (f - 0xD800) * 1024 + w - 0xDC00 + 0x10000;
   } 
};

});
