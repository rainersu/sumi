define(function () {

var d = {},
	e = {},
	a = 'gt=>,lt=<,amp=&,quot=",nbsp= '.split(','),
	l = a.length - 1,
	v;

for(; l >= 0; l--) {
	v = a[l].split('=');
	d[v[0]] = v[1];
	e[v[1]] = v[0];
}

return {
	decode: function(s) {
		return ('' + s).replace(/&([^&;]+);/g, function(a, b) {
			return d[b] || a;
		});
	},
	encode: function(s) {
		return ('' + s).replace(/[><&" ]/g,    function(a, b) {
			return '&' + e[a] + ';';
		});
	}
};

});
