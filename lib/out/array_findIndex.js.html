define(function () {

return Array.prototype.findIndex = Array.prototype.findIndex || function(func, that) {
	for(var i = 0, o = Object(this), l = o.length >>> 0, v; i < l; i++) {
		if (i in o && func.call(that, v = o[i], i, o)) return i;
	}
	return -1;	
};

});
