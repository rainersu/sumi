define(function () {

return Array.prototype.indexOf = Array.prototype.indexOf || function (searchElement, fromIndex) {
	var o = Object(this),
		l = o.length >>> 0,
		n = fromIndex >> 0,
		k = Math.max(n < 0 ? l + n : n, 0),
		s = searchElement,
		i;
    while(k < l) { 
		if(k in o) {
			i = o[k];
			if (s === i || (s !== s && i !== i)) {
				return k;
			}
		}
		k++;
	}
    return -1;
};

});
