define(function () {

return Array.prototype.forEach = Array.prototype.forEach || function(func, that) {
	for(var i = 0, t = this, o = Object(t), l = o.length >>> 0; i < l; i++) {
		if (i in o) func.call(that, o[i], i, o);
    }
	return t;
};

});
