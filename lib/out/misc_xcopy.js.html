return (function f(dst, src, ovr, dpc) {
	var s = Object(src),
		d = Object(dst),
		r = ovr,
		c = dpc,
		i,
		m,
		t,
		b;
	for(i in s) {
		if (H(s, i)) {
			if (r || !H(d, i)) {
				m = s[i];
				t = C(m);
				d[i] = c && (b = t == 'object' ? {} : t == 'array' ? [] : 0) ? f(b, m, r, c) : m;
			}
		}
	}
	return d;
});

});
